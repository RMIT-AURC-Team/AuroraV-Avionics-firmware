114void BMP581_writeRegister(BMP581 *baro, uint8_t address, uint8_t data) {
115 uint16_t response;
116 SPI spi = baro->base;
117
118 // Send write command with address and data
119 uint16_t payload = (address << 0x08) | data;
120 spi.port->ODR &= ~spi.cs;
121 spi.send(&spi, payload);
122
123 // Read in response from interface
124 spi.receive(&spi, &response);
