71void write_lora_packet(uint8_t address, uint8_t payload) {
72 uint16_t return_value = 0; // clear varibles
73 uint16_t payload_to_send = 0;
74 payload_to_send &= (~(0XFFFF)); // clear variable
75 payload_to_send |= (address << 0x8); // load address into top 7 bits
76 payload_to_send |= (1 << 0xF); // set rw bit to write
77 payload_to_send |= payload; // mask in data
78 GPIOD->ODR &= (~(GPIO_ODR_OD0)); // lower chip select
79 while ((SPI3->SR & SPI_SR_TXE) == 0); // wait for transmission to be empty
80 SPI3->DR = payload_to_send; // load data reg
81 while ((SPI3->SR & SPI_SR_RXNE) == 0); // wait for received data
82 return_value = (uint16_t)(SPI3->DR); // see if it returns anything
83 while ((SPI3->SR & SPI_SR_BSY) == 1); // wait for line to be clear
84 GPIOD->ODR |= (GPIO_ODR_OD0); // raise chip select
85}
