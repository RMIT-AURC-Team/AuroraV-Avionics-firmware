42void configure_SPI4_Flash(void) {
43 GPIOE->MODER &= (~(GPIO_MODER_MODE12_Msk | GPIO_MODER_MODE13_Msk | GPIO_MODER_MODE14_Msk));
44 GPIOE->MODER |= ((0x2 << GPIO_MODER_MODE12_Pos) | (0x2 << GPIO_MODER_MODE13_Pos) | (0x2 << GPIO_MODER_MODE14_Pos));
45 GPIOE->AFR[1] &= (uint32_t)(~(0x0FFFF000)); // clears AFRH 11, 12, 13 and 14
46 GPIOE->AFR[1] |= (0x05555000); // sets AFRH 10, 11 and 12 to AF6 for lora SPI
47 GPIOE->MODER &= (~(GPIO_MODER_MODE9_Msk | GPIO_MODER_MODE10_Msk | GPIO_MODER_MODE11_Msk));
48 GPIOE->MODER |= ((0x1 << GPIO_MODER_MODE9_Pos) | (0x1 << GPIO_MODER_MODE10_Pos) | (0x1 << GPIO_MODER_MODE11_Pos));
49 GPIOE->OTYPER &= (uint16_t)(~(GPIO_OTYPER_OT9 | GPIO_OTYPER_OT10 | GPIO_OTYPER_OT11));
50 GPIOE->OSPEEDR &= (~(GPIO_OSPEEDR_OSPEED9_Msk | GPIO_OSPEEDR_OSPEED10_Msk | GPIO_OSPEEDR_OSPEED11_Msk));
51 GPIOE->OSPEEDR |= ((0x2 << GPIO_OSPEEDR_OSPEED9_Pos) | (0x2 << GPIO_OSPEEDR_OSPEED10_Pos) | (0x2 << GPIO_OSPEEDR_OSPEED11_Pos));
52 GPIOE->OTYPER &= (~(GPIO_OTYPER_OT12 | GPIO_OTYPER_OT13 | GPIO_OTYPER_OT14));
53 GPIOE->OSPEEDR &= (~(GPIO_OSPEEDR_OSPEED12_Msk | GPIO_OSPEEDR_OSPEED13_Msk | GPIO_OSPEEDR_OSPEED14_Msk));
54 GPIOE->OSPEEDR |= (0x2 << GPIO_OSPEEDR_OSPEED12_Pos | 0x2 << GPIO_OSPEEDR_OSPEED13_Pos | 0x2 << GPIO_OSPEEDR_OSPEED14_Pos);
55 GPIOE->ODR |= (GPIO_ODR_OD10) | (GPIO_ODR_OD11) | (GPIO_ODR_OD9);
56 SPI4->CR1 &= (~(SPI_CR1_BR_Msk));
57 SPI4->CR1 &= (~(SPI_CR1_CPHA_Msk) | (SPI_CR1_CPOL_Msk));
58 SPI4->CR1 |= SPI_CR1_MSTR; // micro is master
59 SPI4->CR1 |= SPI_CR1_SSM | SPI_CR1_SSI; // Software management
60 SPI4->CR1 &= (~(SPI_CR1_LSBFIRST_Msk)); // MSB FIRST
61 SPI4->CR1 &= ~(SPI_CR1_BIDIMODE | SPI_CR1_RXONLY);
62 SPI4->CR1 &= (~(SPI_CR1_BR_Msk));
63 SPI4->CR1 |= ((0x01 << SPI_CR1_BR_Pos));
64 SPI4->CR1 |= (0x1 << SPI_CR1_SPE_Pos);
65}

71void configure_UART3_GPS(void) {
72 GPIOD->MODER &= (~(GPIO_MODER_MODE8_Msk | GPIO_MODER_MODE9_Msk | GPIO_MODER_MODE13_Msk));
73 GPIOD->MODER |= ((0x2 << GPIO_MODER_MODE8_Pos) | (0x2 << GPIO_MODER_MODE9_Pos) | (0x1 << GPIO_MODER_MODE13_Pos));
74 GPIOD->AFR[1] &= (uint32_t)(~(0x000000FF)); // clears AFRL 6 and 7
75 GPIOD->AFR[1] |= (0x00000077); // sets PD 8, 9 and 13 to AF7
76
77 GPIOD->PUPDR |= (0X1 << GPIO_PUPDR_PUPD9_Pos);
78
79 GPIOD->OSPEEDR &= (~(GPIO_OSPEEDR_OSPEED8 | GPIO_OSPEEDR_OSPEED9 | GPIO_OSPEEDR_OSPEED13));
80 GPIOD->OSPEEDR |= (0x3 << GPIO_OSPEEDR_OSPEED8_Pos) | (0x3 << GPIO_OSPEEDR_OSPEED9_Pos) | (0x3 << GPIO_OSPEEDR_OSPEED13_Pos);
81
82 // need over sampling = 1
83 USART3->BRR &= (unsigned int)(0xFFFF0000); // clear mantissa and div in baud rate reg
84 USART3->BRR |= (0x0002227); // set mantissa and div in baud rate reg to 9600
85
86 USART3->CR1 &= (unsigned int)(~(0x400)); // disable parity
87 USART3->CR2 &= (unsigned int)(~(0xE00)); // disable synchrnous mode
88 USART3->CR3 &= (unsigned int)(~(0x300)); // disable flow control
89 USART3->CR1 |= (unsigned int)(0x200C); // enable usart, enable receive and transmitt
90 USART3->CR1 |= USART_CR1_OVER8;
91
92 // turn reset pin high
93 GPIOD->ODR |= GPIO_ODR_OD13;
94}

96// General GPIO Configure for MISC: LED1, LED2, Piezo Buzzer (PD14, PD15, PB15 respectivley)
97void configure_MISC_GPIO(void) {
98 GPIOD->MODER &= (~(GPIO_MODER_MODE14_Msk | GPIO_MODER_MODE15_Msk));
99 GPIOD->MODER |= ((0x1 << GPIO_MODER_MODE14_Pos) | (0x1 << GPIO_MODER_MODE15_Pos));
100 GPIOD->OTYPER &= (uint16_t)(~(GPIO_OTYPER_OT14 | GPIO_OTYPER_OT15)); // sets 0xboth as push-pull
101 GPIOD->OSPEEDR &= (~(GPIO_OSPEEDR_OSPEED14_Msk | GPIO_OSPEEDR_OSPEED15_Msk)); // clears Port 14 and 15 section
102 GPIOD->OSPEEDR |= ((0x2 << GPIO_OSPEEDR_OSPEED14_Pos) | (0x2 << GPIO_OSPEEDR_OSPEED15_Pos)); // sets slew rate as high speed
103
104 // GPIOD->PUPDR &= (~(GPIO_PUPDR_PUPD14 | GPIO_PUPDR_PUPD15)); // clears register for port 14 and 15
105 // GPIOD->PUPDR |= (0x00<<GPIO_PUPDR_PUPD14_Pos | 0x00<<GPIO_PUPDR_PUPD15_Pos ); // shifts 00 into and pos for 14 and 15
106 GPIOD->ODR &= (~(GPIO_ODR_OD14 | GPIO_ODR_OD15)); // turns LEDs off
107
108 GPIOB->MODER &= (~(GPIO_MODER_MODE15_Pos)); // clears pos 15 of port B moder R reg
109 GPIOB->MODER |= (0x1 << GPIO_MODER_MODE15_Pos); // sets pos 15 to general purpose output
110 GPIOB->OTYPER &= (~(GPIO_OTYPER_OT15_Msk)); // sets port B 15 to push-pull
111 GPIOB->OSPEEDR &= (~(GPIO_OSPEEDR_OSPEED15_Msk)); // clears pos 15 in Ospeed R reg
112 GPIOB->OSPEEDR |= (0x2 << GPIO_OSPEEDR_OSPEED15_Pos); // sets slew rate to highspeed
113 GPIOB->ODR &= (~(GPIO_ODR_OD15)); // turns peizer buzzer IO to low
114
115 // configure PB1 as input for B switch
116 GPIOB->MODER &= (~(GPIO_MODER_MODE1_Pos));
117 GPIOB->OTYPER &= (~(GPIO_OTYPER_OT1_Msk)); // sets port B 15 to push-pull
118 GPIOB->OSPEEDR &= (~(GPIO_OSPEEDR_OSPEED1_Msk)); // clears pos 15 in Ospeed R reg
119 GPIOB->OSPEEDR |= (0x2 << GPIO_OSPEEDR_OSPEED1_Pos); // sets slew rate to highspeed
120 GPIOB->PUPDR &= (~(GPIO_PUPDR_PUPD1)); // clears PUPDR for PB1
121 GPIOB->PUPDR |= (0x01 << GPIO_PUPDR_PUPD1_Pos); // set pull up resistor for PB1
122}

160void buzzer(void) {
161 TIM6->ARR &= (~(TIM_ARR_ARR_Msk));
162 TIM6->PSC &= (~(TIM_PSC_PSC_Msk));
163 TIM6->ARR |= 80;
164 TIM6->PSC |= 128;
165 TIM7->CR1 |= TIM_CR1_CEN;
166 TIM6->CR1 |= TIM_CR1_CEN; // ensures timer is enabled
167 while ((TIM7->SR & TIM_SR_UIF) == 0) {
168 GPIOB->ODR ^= 0X8000;
169 while ((TIM6->SR & TIM_SR_UIF) == 0);
170 TIM6->SR &= ~(TIM_SR_UIF); // clears UIF
171 TIM6->ARR |= 80;
172 TIM6->CR1 |= TIM_CR1_CEN; // Enables counter
173 }
174 TIM7->SR &= ~(TIM_SR_UIF);
