115void Flash_erase(Flash *flash) {
116 _Flash_writeEnable(flash);
117 SPI spi = flash->base;
118 uint8_t status = 0;
119
120 // Send Erase Chip instruction
121 spi.port->ODR &= ~spi.cs;
122 spi.transmit(&spi, FLASH_ERASE_CHIP);
123 spi.port->ODR |= spi.cs;
124
125 // Wait until chip BUSY is clear

127 _Flash_readStatus1(flash, &status);

140void Flash_writePage(Flash *flash, uint32_t address, uint8_t *data) {
141 _Flash_writeEnable(flash);
142 uint8_t status = 0;
143 SPI spi = flash->base;
144
145 spi.port->ODR &= ~spi.cs;
146
147 // Send Page Program instruction and 24-bit address
148 spi.transmit(&spi, FLASH_PAGE_PROGRAM);
149 spi.transmit(&spi, (address & 0xFF0000) >> 16);
150 spi.transmit(&spi, (address & 0xFF00) >> 8);
151 spi.transmit(&spi, (address & 0xFF));
152
153 // Send page data
154 for (int x = 0; x < 255; x++) {
155 spi.transmit(&spi, data[x]);
156 }
157
158 spi.port->ODR |= spi.cs;
159
160 // Wait until chip BUSY is clear
161 do {
162 _Flash_readStatus1(flash, &status);
163 } while (status & 0x1);
164}
