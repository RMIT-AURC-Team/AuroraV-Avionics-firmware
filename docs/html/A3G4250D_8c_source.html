69void A3G4250D_processRawBytes(A3G4250D *gyro, uint8_t *bytes, float *out) {
70 out[0] = gyro->sign[0] * gyro->sensitivity * (int16_t)(((uint16_t)bytes[0] << 8) | bytes[1]); // gyro X
71 out[1] = gyro->sign[1] * gyro->sensitivity * (int16_t)(((uint16_t)bytes[2] << 8) | bytes[3]); // gyro Y
72 out[2] = gyro->sign[2] * gyro->sensitivity * (int16_t)(((uint16_t)bytes[4] << 8) | bytes[5]); // gyro Z
73}

97 uint16_t response;
98 SPI spi = gyro->base;
99
100 // Send write command with address and data
101 uint16_t payload = (address << 0x08) | data;
102 spi.port->ODR &= ~spi.cs;
