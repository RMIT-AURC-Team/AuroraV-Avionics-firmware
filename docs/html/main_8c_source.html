127void vStateUpdate(void *argument) {
128 unsigned int ms = 0;
129 union {
130 unsigned int i;
131 uint8_t a[4];
132 } u;
133 TickType_t xLastWakeTime;
134 const TickType_t xFrequency = pdMS_TO_TICKS(20); // 50Hz
135
136 unsigned int CANHigh = 0;
137 unsigned int CANLow = 0;
138 unsigned int id = 0;
139
140 for (;;) {
141 // Block until 20ms interval
142 vTaskDelayUntil(&xLastWakeTime, xFrequency);
143
144 // Emergency aerobrakes for excessive tilt
145 if (tilt >= 30.0f) {
146 // CAN payload for aerobrakes retract
147 CANHigh = 0x00000000;
148 CANLow = 0x00000000;
149 id = 0x602;
150 CAN_TX(1, 8, CANHigh, CANLow, id);
151 }
152
153 switch (currentState) {
154 case PRELAUNCH:
155 if (accel_s.accelData[ZINDEX] >= ACCEL_LAUNCH) {
156 xEventGroupSetBits(xTaskEnableGroup, 0x80); // Enable flash
157 xEventGroupSetBits(xTaskEnableGroup, 0x06); // Enable data acquisition
158 currentState = LAUNCH;
159 // Add launch event dataframe to buffer
160 mem.append(&mem, HEADER_EVENT_LAUNCH);
161 mem.appendBytes(&mem, u.a, 4);
162 }
163 break;
164 case LAUNCH:
165 // Send altitude to aerobrakes via CAN
166 CANHigh = 0x00000000;
167 CANLow = (unsigned int)altitude;
168 id = 0x601;
169 CAN_TX(1, 8, CANHigh, CANLow, id);
170 // Transition to motor burnout state on velocity decrease
171 if (true) { // TODO: Change to decreasing velocity
172 currentState = COAST;
173 // Add motor burnout event dataframe to buffer
174 mem.append(&mem, HEADER_EVENT_COAST);
175 mem.appendBytes(&mem, u.a, 4);
176 }
177 break;
178 case COAST:
179 // Send altitude to aerobrakes via CAN
180 CANHigh = 0x00000000;
181 CANLow = (unsigned int)altitude;
182 id = 0x601;
183 CAN_TX(1, 8, CANHigh, CANLow, id);
184 // Transition to apogee state on three way vote of altitude, velocity, and tilt
185 // apogee is determined as two of three conditions evaluating true
186 if (((true) + (tilt >= 90) + (velocity < 0.0f)) >= 2) { // TODO: change first condition to decreasing altitude
187 currentState = APOGEE;
188 // Add apogee event dataframe to buffer
189 mem.append(&mem, HEADER_EVENT_APOGEE);
190 mem.appendBytes(&mem, u.a, 4);
191 // Send transmission to trigger apogee E-matches
192 }
193 break;
194 case APOGEE:
195 if (isAltitude1300ft(altitude)) {
196 currentState = DESCENT;
197 // Enable descent state
198 mem.append(&mem, HEADER_EVENT_DESCENT);
199 mem.appendBytes(&mem, u.a, 4);
200 // Add descent event dataframe to buffer
201 }
202 break;
203 case DESCENT:
204 // Handle descent state actions
205 break;
206 }
207
208 if(currentState >= LAUNCH) {
209 LoRa_Packet packet = LoRa_AVD1(LORA_HEADER_AVD1, accel_s.rawAccelData, KX134_1211_DATA_TOTAL);
210 xMessageBufferSend(xLoRaTxBuff, (void *) &packet, sizeof(packet), 0);
211 }
212
213 ms += 2;
214 u.i = ms;
215 }

286void vDataAcquisitionH(void *argument) {
287 float dt = 0.002;
288
289 TickType_t xLastWakeTime;
290 const TickType_t xFrequency = pdMS_TO_TICKS(2); // 500Hz
291 const TickType_t blockTime = pdMS_TO_TICKS(0); // Don't need to block calculations
292 for (;;) {
293 // Block until 2ms interval
294 vTaskDelayUntil(&xLastWakeTime, xFrequency);
295
296 // Select which accelerometer to use
297 accel_s = (accel_s.accelData[ZINDEX] < 15) ? lAccel_s : hAccel_s;
298
299 // Update sensor data
300 accel_s.update(&accel_s);
301 gyro_s.update(&gyro_s);
302
303 // Add sensor data to dataframe
304 mem.append(&mem, HEADER_HIGHRES);
305 mem.appendBytes(&mem, accel_s.rawAccelData, KX134_1211_DATA_TOTAL);
306 mem.appendBytes(&mem, gyro_s.rawGyroData, A3G4250D_DATA_TOTAL);
307
308 // Only run calculations when enabled
309 EventBits_t uxBits = xEventGroupWaitBits(xTaskEnableGroup, 0x02, pdFALSE, pdFALSE, blockTime);
310 if ((uxBits & 0x02) == 0x02) {
311 // Integrate attitude quaternion from rotations
312 Quaternion qDot;
313 Quaternion_init(&qDot);
314 qDot.fromEuler(
315 &qDot,
316 (float)(dt * gyro_s.gyroData[ROLL_INDEX]),
317 (float)(dt * gyro_s.gyroData[PITCH_INDEX]),
318 (float)(dt * gyro_s.gyroData[YAW_INDEX])
319 );
320 qRot = Quaternion_mul(&qRot, &qDot);
321 qRot.normalise(&qRot); // New attitude quaternion
322
323 // Apply rotation to z-axis unit vector
324 qRot.fRotateVector3D(&qRot, zUnit, vAttitude);
325
326 // Calculate tilt angle
327 // tilt = cos^-1(attitude initial)
328 cosine = zUnit[0] * vAttitude[0] + zUnit[1] * vAttitude[1] + zUnit[2] * vAttitude[2];
329 tilt = acos(cosine) * 180 / M_PI;
330 }
331 }
332}

338void vDataAcquisitionL(void *argument) {
339 float dt = 0.020;
340 KalmanFilter kf;
341 KalmanFilter_init(&kf);
342
343 // Initialise filter parameters
344 float A[9] = {
345 1.0, dt, 0.5 * (dt * dt),
346 0.0, 1.0, dt,
347 0.0, 0.0, 1.0
348 };
349 kf.A.pData = A;
350 float Q[9] = {
351 99.52, 0.0, 0.0,
352 0.0, 1.42, 0.0,
353 0.0, 0.0, 6.27
354 };
355 kf.Q.pData = Q;
356 float R[4] = {
357 97.92, 0.0,
358 0.0, 0.61
359 };
360 kf.R.pData = R;
361 float P[9] = {
362 1, 0.0, 0.0,
363 0.0, 0.1, 0.0,
364 0.0, 0.0, 100.0
365 };
366 kf.P.pData = P;
367
368 // Initialise measurement matrix
369 arm_matrix_instance_f32 z;
370 float zData[2] = {0.0, 0.0};
371 arm_mat_init_f32(&z, 2, 1, zData);
372
373 TickType_t xLastWakeTime;
374 const TickType_t xFrequency = pdMS_TO_TICKS(20); // 50Hz
375 const TickType_t blockTime = pdMS_TO_TICKS(0); // Don't need to block
376
377 for (;;) {
378 // Block until 20ms interval
379 vTaskDelayUntil(&xLastWakeTime, xFrequency);
380
381 // Update baro data
382 baro_s.update(&baro_s);
383
384 // Calculate altitude
385 altitude = 44330 * (1.0 - pow(baro_s.press / baro_s.groundPress, 0.1903));
386
387 // Add sensor data and barometer data to dataframe
388 mem.append(&mem, HEADER_LOWRES);
389 mem.appendBytes(&mem, baro_s.rawTemp, 3);
390 mem.appendBytes(&mem, baro_s.rawPress, 3);
391
392 // Only run calculations when enabled
393 EventBits_t uxBits = xEventGroupWaitBits(xTaskEnableGroup, 0x04, pdFALSE, pdFALSE, blockTime);
394 if ((uxBits & 0x04) == 0x04) {
395 // Calculate state
396 z.pData[0] = altitude;
397 z.pData[1] = (cosine * 9.81 * accel_s.accelData[ZINDEX] - 9.81); // Acceleration measured in m/s^2
398 kf.update(&kf, &z);
399 }
400 }
