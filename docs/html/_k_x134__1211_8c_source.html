14void KX134_1211_init(KX134_1211 *accel, GPIO_TypeDef *port, unsigned long cs, uint8_t scale, const uint8_t *axes) {
15 SPI_init(&accel->base, SENSOR_ACCEL, SPI1, port, cs);
16 accel->axes = axes;
17 accel->readAccel = KX134_1211_readAccel;
18 accel->readRawBytes = KX134_1211_readRawBytes;
19 accel->processRawBytes = KX134_1211_processRawBytes;
20
21 // Set value of GSEL and sensitivity based on selected scale
22 uint8_t GSEL = 0x00;
23 if (scale == 32) {
24 GSEL = KX134_1211_CNTL1_GSEL(32);
25 accel->sensitivity = KX134_1211_SENSITIVITY(32);
26 } else if (scale == 16) {
27 GSEL = KX134_1211_CNTL1_GSEL(16);
28 accel->sensitivity = KX134_1211_SENSITIVITY(16);
29 }
30
31 // Configure accelerometer registers
32 KX134_1211_writeRegister(accel, KX134_1211_CNTL1, KX134_1211_CNTL1_RES | GSEL); // Accel select, selected sensitivity
33 uint8_t ODCNTL = KX134_1211_readRegister(accel, KX134_1211_ODCNTL); // Read from register for reserve mask
34 KX134_1211_writeRegister(accel, KX134_1211_ODCNTL, (KX134_1211_ODCNTL_RESERVED & ODCNTL) | 0x2A); // No filter, fast startup, 800Hz
35 KX134_1211_writeRegister(accel, KX134_1211_CNTL1, KX134_1211_CNTL1_PC1 | KX134_1211_CNTL1_RES | GSEL); // Enable PC1
36}

63void KX134_1211_processRawBytes(KX134_1211 *accel, uint8_t *bytes, float *out) {
64 out[0] = accel->sensitivity * (int16_t)(((uint16_t)bytes[0] << 8) | bytes[1]); // Accel X
65 out[1] = accel->sensitivity * (int16_t)(((uint16_t)bytes[2] << 8) | bytes[3]); // Accel Y
66 out[2] = accel->sensitivity * (int16_t)(((uint16_t)bytes[4] << 8) | bytes[5]); // Accel Z
67}

88void KX134_1211_writeRegister(KX134_1211 *accel, uint8_t address, uint8_t data) {
89 uint16_t response;
90 SPI spi = accel->base;
91
92 // Send write command with address and data
93 uint16_t payload = (address << 0x08) | data; // Load payload with address and data
94 spi.port->ODR &= ~spi.cs; // Lower chip select
95 spi.send(&spi, payload); // Send payload
96
97 // Read in response from interface
98 spi.receive(&spi, &response); // Read in response from receive buffer
99 spi.port->ODR |= spi.cs; // Raise chip select
100}
