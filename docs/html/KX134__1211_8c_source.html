19void KX134_1211_init(KX134_1211 *accel, GPIO_TypeDef *port, unsigned long cs, uint8_t scale, const uint8_t *axes, const int8_t *sign) {
20 SPI_init(&accel->base, SENSOR_ACCEL, SPI1, port, cs);
21 accel->update = KX134_1211_update;
22 accel->readAccel = KX134_1211_readAccel;
23 accel->readRawBytes = KX134_1211_readRawBytes;
24 accel->processRawBytes = KX134_1211_processRawBytes;
25 memcpy(accel->axes, axes, KX134_1211_DATA_COUNT);
26 memcpy(accel->sign, sign, KX134_1211_DATA_COUNT);
27
28 // Set value of GSEL and sensitivity based on selected scale
29 uint8_t GSEL = 0x00;
30 if (scale == 32) {
31 GSEL = KX134_1211_CNTL1_GSEL(32);
32 accel->sensitivity = KX134_1211_SENSITIVITY(32);
33 } else if (scale == 16) {
34 GSEL = KX134_1211_CNTL1_GSEL(16);
35 accel->sensitivity = KX134_1211_SENSITIVITY(16);
36 }
37
38 // Configure accelerometer registers
39 KX134_1211_writeRegister(accel, KX134_1211_CNTL1, KX134_1211_CNTL1_RES | GSEL); // Accel select, selected sensitivity
40 uint8_t ODCNTL = KX134_1211_readRegister(accel, KX134_1211_ODCNTL); // Read from register for reserve mask
41 KX134_1211_writeRegister(accel, KX134_1211_ODCNTL, (KX134_1211_ODCNTL_RESERVED & ODCNTL) | 0x2A); // No filter, fast startup, 800Hz
42 KX134_1211_writeRegister(accel, KX134_1211_CNTL1, KX134_1211_CNTL1_PC1 | KX134_1211_CNTL1_RES | GSEL); // Enable PC1
43}

81void KX134_1211_processRawBytes(KX134_1211 *accel, uint8_t *bytes, float *out) {
82 out[0] = accel->sign[0] * accel->sensitivity * (int16_t)(((uint16_t)bytes[0] << 8) | bytes[1]); // Accel X
83 out[1] = accel->sign[1] * accel->sensitivity * (int16_t)(((uint16_t)bytes[2] << 8) | bytes[3]); // Accel Y
84 out[2] = accel->sign[2] * accel->sensitivity * (int16_t)(((uint16_t)bytes[4] << 8) | bytes[5]); // Accel Z
85}

95void KX134_1211_readRawBytes(KX134_1211 *accel, uint8_t *out) {

99 out[INDEX_AXES(0, 1)] = KX134_1211_readRegister(accel, KX134_1211_XOUT_L); // Accel X low

102 out[INDEX_AXES(2, 0)] = KX134_1211_readRegister(accel, KX134_1211_ZOUT_H); // Accel Z high
103 out[INDEX_AXES(2, 1)] = KX134_1211_readRegister(accel, KX134_1211_ZOUT_L); // Accel Z low
104#undef INDEX_AXES

110 uint16_t response;
111 SPI spi = accel->base;
112
113 // Send write command with address and data
114 uint16_t payload = (address << 0x08) | data; // Load payload with address and data
115 spi.port->ODR &= ~spi.cs; // Lower chip select
116 spi.send(&spi, payload); // Send payload

120 spi.port->ODR |= spi.cs; // Raise chip select
