122void LoRa_writeRegister(LoRa *lora, uint8_t address, uint8_t data) {
123 SPI spi = lora->base;
124
125 uint16_t payload = (address << 0x08) | (1 << 0x0F); // Load payload with address and write command

127 spi.port->ODR &= ~spi.cs; // Lower chip select
128 spi.transmit(&spi, payload); // Send payload over SPI
129 spi.port->ODR |= spi.cs; // Raise chip select
